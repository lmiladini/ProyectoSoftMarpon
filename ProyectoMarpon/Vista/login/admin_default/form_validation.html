Default validators

The default features of this plugin doesn't require you to load any module. Here you can try out a complete form with all default validators present.

<form action="#" method="POST">
    <label>Username</label>
    <div class="form-group">
        <input type="text" class="form-control" data-validation="required" placeholder="Username">
    </div>

    <div class="row">
        <div class="col-xl-6">
            <label>Minimum length 20</label>
            <div class="form-group">
                <input class="form-control" data-validation="length" data-validation-length="min20" type="text" placeholder="Minimum length 20">
            </div>
        </div>
        <div class="col-xl-6">
            <label>Length between 50-100</label>
            <div class="form-group">
                <input class="form-control" data-validation="length" data-validation-length="50-100" type="text" placeholder="Length between 50-100">
            </div>
        </div>
    </div>

    <label>Only number input</label>
    <div class="form-group">
        <input class="form-control" data-validation="number" type="text" placeholder="Only number input">
    </div>

    <div class="row">
        <div class="col-xl-6">
            <label>Enter email</label>
            <div class="form-group">
                <input class="form-control" data-validation="email" type="text" placeholder="Enter Email">
            </div>
        </div>
        <div class="col-xl-6">
            <label>Enter url</label>
            <div class="form-group">
                <input class="form-control" data-validation="url" type="text" placeholder="Enter url">
            </div>
        </div>
    </div>

    <label>File type</label>
    <div class="form-group file-control-upload">
        <input class="form-control-file" type="file" multiple="multiple" name="images" data-validation="length mime size" data-validation-length="min2" data-validation-allowing="jpg, png, gif" data-validation-max-size="512kb" data-validation-error-msg-size="You can not upload images larger than 512kb" data-validation-error-msg-mime="You can only upload images" data-validation-error-msg-length="You have to upload at least two images">
    </div>

    <label>Enter alphanumeric characters</label>
    <div class="form-group">
        <input class="form-control" data-validation="alphanumeric" type="text" placeholder="Enter alphanumeric">
    </div>

    <div class="row">
        <div class="col-xl-6">
            <label>Enter alphanumeric characters and -_</label>
            <div class="form-group">
                <input class="form-control" data-validation="alphanumeric" data-validation-allowing="-_" type="text" placeholder="Enter alphanumeric and -_">
            </div>
        </div>
        <div class="col-xl-6">
            <label>Enter lowercase letters a-z</label>
            <div class="form-group">
                <input class="form-control" data-validation="custom" data-validation-regexp="^([a-z]+)$" type="text" placeholder="Enter lowercase letters a-z">
            </div>
        </div>
    </div>

    <div class="form-group">
        <label>History (<span id="maxlength">50</span> characters left)</label>
        <textarea class="form-control" rows="3" id="area" placeholder="Enter lowercase letters a-z"></textarea>
    </div>

    <div class="form-group">
        <button type="submit" class="btn btn-primary site-btn">Submit</button>
    </div>
</form>

Security validators

This module contains validators commonly used in registration forms. Here you can try an example form with all the security validators in action.

<form action="#" id="form">
    <label>Username</label>
    <div class="form-group">
        <input class="form-control" name="user" type="text" data-validation="length alphanumeric" data-validation-length="3-12" data-validation-error-msg="Between 3-12 chars, only alphanumeric characters" placeholder="Enter Username.">
    </div>

    <label>Password</label>
    <div class="form-group">
        <input class="form-control" name="pass_confirmation" type="password" data-validation="strength" data-validation-strength="2" placeholder="Enter Password.">
    </div>

    <label>Confirm Password</label>
    <div class="form-group">
        <input class="form-control" name="pass" type="password" data-validation="confirmation" placeholder="Enter confirm Password.">
    </div>

    <label>Credit card</label>
    <div class="form-group">
        <select name="ccard" id="card" class="form-control">
            <option value="visa">VISA</option>
            <option value="mastercard">Mastercard</option>
            <option value="amex">American express</option>
            <option value="diners_club">Diners club</option>
            <option value="discover">Discover</option>
            <option value="cjb">CJB</option>
            <option value="maestro">Maestro</option>
        </select>
    </div>

    <label>Credit card number</label>
    <div class="form-group">
        <input class="form-control" name="ccard_num" data-validation="creditcard" data-validation-allowing="visa" placeholder="Enter Credit card number.">
    </div>

    <label>CVV</label>
    <div class="form-group">
        <input class="form-control" name="cvv" data-validation="cvv" placeholder="Enter CVV number.">
    </div>

    <label>What is the sum of 3 + 4?</label>
    <div class="form-group">
        <input class="form-control" data-validation="spamcheck" data-validation-captcha="7" placeholder="Enter Sum.">
    </div>

    <label>Only letters, numbers and spaces allowed</label>
    <div class="form-group">
        <input class="form-control" data-validation="letternumeric" data-validation-allowing=" " type="text" placeholder="Enter text.">
    </div>

    <div class="form-group">
        <button type="submit" class="btn btn-primary site-btn">Submit</button>
    </div>
</form>

Date validators

This module contains validators needed when validating dates and time. Here you can try an example form with all the date validators in action.

<form action="#" id="date-form">
    <label>Birth date (yyyy-mm-dd)</label>
    <div class="form-group">
        <input class="form-control" data-validation="birthdate" placeholder="Enter Birth date">
    </div>

    <label>Time (HH:mm)</label>
    <div class="form-group">
        <input class="form-control" data-validation="time" data-validation-help="HH:mm" placeholder="Enter time HH:mm">
    </div>

    <label>Birth date (dd/mm/yyyy)</label>
    <div class="form-group">
        <input class="form-control" data-validation="birthdate" data-validation-format="dd/mm/yyyy" placeholder="Enter Birth date">
    </div>

    <div class="form-group">
        <button type="submit" class="btn btn-primary site-btn">Submit</button>
    </div>
</form>

Validation states

To use validation states, add class .success-input, .info-input, .warning-input or .danger-input.

<label>Success field</label>
<div class="form-group">
    <div class="input-group input-right-icon success-input">
        <input class="form-control" type="text" placeholder="Success Field">
        <span class="input-group-addon"><i class="fa fa-check"></i></span>
    </div>
</div>
